#### 1.3 Control Flow Architecture

Research on coding agent architectures shows that **imperative while-loops** (used by Aider, OpenCode, Gemini CLI) are superior to recursive approaches (used by Cline) for state serialization and checkpointing .

```python
# ReAct Agent Loop (imperative while-loop pattern)
while not task_complete:
    # 1. Build context window from state + retrieved code
    context = build_context(conversation_history, repo_map, user_input)
    
    # 2. Send to LLM via Anthropic-compatible API
    response = llm_client.messages.create(
        model=current_model,
        messages=context,
        tools=available_tools,
        max_tokens=8192
    )
    
    # 3. Parse response for tool calls
    if response.stop_reason == "tool_use":
        for tool_call in response.content:
            # 4. Execute tool with safety checks
            result = tool_registry.execute(tool_call, approval_policy)
            
            # 5. Append result to conversation state
            conversation_history.append({
                "role": "user",
                "content": [{"type": "tool_result", ...}]
            })
    else:
        # Final answer
        render_streaming_output(response.content)
        task_complete = True
```

### 2. API Compatibility: Native Anthropic Messages Format

#### 2.1 Request Structure

```json

    {
      "role": "user",

  "tools": [

      "name": "read_file",

        "type": "object",
```

```python
import os
from dataclasses import dataclass

@dataclass
class ProviderConfig:
    base_url: str
    api_key: str
    model_map: dict[str, str]
    is_local: bool

class EnvironmentResolver:
    ANTHROPIC_DEFAULTS = {
        'base_url': 'https://api.anthropic.com',
        'api_key': None,
        'model_map': {
            'sonnet': 'claude-sonnet-4-20250514',
            'opus': 'claude-opus-4-20260416',
            'haiku': 'claude-haiku-4-20250514'
        }
    }
    
    @classmethod
    def resolve(cls) -> ProviderConfig:
        base_url = os.getenv('ANTHROPIC_BASE_URL', cls.ANTHROPIC_DEFAULTS['base_url'])
        api_key = os.getenv('ANTHROPIC_API_KEY', '')
        
        # Critical: API key must be non-empty for local mode
        if not api_key and base_url != cls.ANTHROPIC_DEFAULTS['base_url']:
            raise ValueError(
                "ANTHROPIC_API_KEY must be set to a non-empty value "
                "(e.g., 'local') when using ANTHROPIC_BASE_URL."
            )
        
        is_local = base_url != cls.ANTHROPIC_DEFAULTS['base_url']
        
        model_map = {
            'sonnet': os.getenv('ANTHROPIC_DEFAULT_SONNET_MODEL', 
                                 cls.ANTHROPIC_DEFAULTS['model_map']['sonnet']),
            'opus': os.getenv('ANTHROPIC_DEFAULT_OPUS_MODEL',
                             cls.ANTHROPIC_DEFAULTS['model_map']['opus']),
            'haiku': os.getenv('ANTHROPIC_DEFAULT_HAIKU_MODEL',
                              cls.ANTHROPIC_DEFAULTS['model_map']['haiku'])
        }
        
        return ProviderConfig(
            base_url=base_url.rstrip('/'),
            api_key=api_key,
            model_map=model_map,
            is_local=is_local
        )
    
    @classmethod
    def resolve_model(cls, requested: str, config: ProviderConfig) -> str:
        if not config.is_local:
            return requested
            
        mapping = {
            'claude-sonnet': config.model_map['sonnet'],
            'claude-opus': config.model_map['opus'],
            'claude-haiku': config.model_map['haiku'],
            'claude-sonnet-4-20250514': config.model_map['sonnet'],
            'claude-opus-4-20260416': config.model_map['opus'],
        }
        
        return mapping.get(requested, config.model_map['sonnet'])
```

#### 4.3 Performance Optimizations

```python
class StateManager:
    def __init__(self, base_dir: Path):
        self.base_dir = base_dir
        self.db_path = base_dir / 'sessions' / 'active' / f'{self.session_id}.db'
        
        # WAL mode for concurrent reads/writes
        self.conn = sqlite3.connect(self.db_path, check_same_thread=False)
        self.conn.execute('PRAGMA journal_mode=WAL')
        self.conn.execute('PRAGMA synchronous=NORMAL')
        self.conn.execute('PRAGMA cache_size=-64000')  # 64MB cache
        self.conn.execute('PRAGMA temp_store=MEMORY')
        
        # In-memory LRU cache for hot messages
        self.message_cache = LRUCache(maxsize=1000)
        
        # Batch writes for high-frequency operations
        self.write_buffer = []
    
    async def append_message(self, message: Message) -> None:
        self.write_buffer.append(message)
        if len(self.write_buffer) >= 100:
            await self._flush_buffer()
    
    async def _flush_buffer(self) -> None:
        if not self.write_buffer:
            return
        with self.conn:
            self.conn.executemany(
                '''INSERT INTO conversations 
                   (session_id, timestamp, role, content, model, token_count_input, token_count_output)
                   VALUES (?, ?, ?, ?, ?, ?, ?)''',
                [(m.session_id, m.timestamp, m.role, m.content, m.model,
                  m.tokens_in, m.tokens_out) for m in self.write_buffer]
            )
        self.write_buffer.clear()
    
    async def get_context_window(self, max_tokens: int = 128000) -> list[Message]:
        cursor = self.conn.execute(
            '''SELECT * FROM conversations 
               WHERE session_id = ? 
               ORDER BY timestamp DESC 
               LIMIT ?''',
            (self.session_id, 500)
        )
        
        messages = []
        total_tokens = 0
        
        for row in cursor:
            msg = Message.from_row(row)
            msg_tokens = msg.token_count_input + msg.token_count_output
            if total_tokens + msg_tokens > max_tokens:
                break
            messages.append(msg)
            total_tokens += msg_tokens
        
        messages.reverse()
        return messages
```

```python
class ContextRetriever:
    def __init__(self, repo_path: Path, state_dir: Path):
        self.repo_path = repo_path
        self.repo_hash = hash_repo(repo_path)
        self.repomap = RepoMap(repo_path, state_dir / 'repos' / self.repo_hash)
        self.embeddings = EmbeddingStore(state_dir / 'repos' / self.repo_hash / 'embeddings')
        self.symbol_index = SymbolIndex(repo_path)
        self.recent_files = LRUCache(maxsize=50)
    
    async def retrieve(self, query: str, current_file: Optional[Path] = None) -> Context:
        # Strategy 1: Repo-map for structural understanding (Aider-style PageRank)
        ranked_files = self.repomap.get_ranked_files(query, top_k=20)
        
        # Strategy 2: Semantic search for conceptually related code
        semantic_results = await self.embeddings.search(query, top_k=10)
        
        # Strategy 3: Symbol search for precise matches
        symbol_results = self.symbol_index.find_symbols(query)
        
        # Strategy 4: Recent working set
        recent = list(self.recent_files.keys())[:10]
        
        # Merge and deduplicate with scoring
        return self._merge_results(
            repomap=ranked_files,
            semantic=semantic_results,
            symbols=symbol_results,
            recent=recent
        )
```

### 5. Tool Use & The Bridge (Translation Proxy)

#### 5.1 Why You Need a Bridge

Local LLMs (via Ollama/LM Studio) speak **OpenAI's chat completions format**, while Claude Code expects **Anthropic's Messages API**. The formats are incompatible for tool use — you cannot simply point `ANTHROPIC_BASE_URL` at Ollama and expect it to work . A translation proxy is essential for:

- Converting Anthropic `tool_use`/`tool_result` blocks to OpenAI `function_call`/`function` messages
- Handling streaming tool calls
- Managing tool schemas between formats
- Supporting parallel tool use

#### 5.2 Bridge Architecture

```
+-----------------------------------------------------------------+
|                     Translation Proxy                           |
|                                                                 |
|  +-------------+      +--------------+      +-------------+    |
|  |  Anthropic  |      |   Request    |      |   OpenAI    |    |
|  |  Messages   |----->|  Transformer |----->|  Chat Comp  |    |
|  |  (Inbound)  |      |              |      |  (Outbound) |    |
|  +-------------+      +--------------+      +-------------+    |
|         |                      ^                   |             |
|         |                      |                   |             |
|         v                      |                   v             |
|  +-------------+      +--------------+      +-------------+    |
|  |  Response   |      |   Response   |      |  LLM        |    |
|  |  Formatter |<-----|  Transformer |<-----|  (Ollama)   |    |
|  |  (Outbound)|      |              |      |             |    |

|                                                                 |
```

#### 5.3 Format Translation Logic

```python
class AnthropicToOpenAITranslator:
    def translate_request(self, anthropic_request: dict) -> dict:
        messages = []
        
        if 'system' in anthropic_request:
            messages.append({
                'role': 'system',
                'content': anthropic_request['system']
            })
        
        for msg in anthropic_request['messages']:
            messages.append(self._convert_message(msg))
        
        tools = None
        if 'tools' in anthropic_request:
            tools = [self._convert_tool(t) for t in anthropic_request['tools']]
        
        return {
            'model': self._map_model(anthropic_request['model']),
            'messages': messages,
            'tools': tools,
            'tool_choice': 'auto' if tools else None,
            'max_tokens': anthropic_request.get('max_tokens', 4096),
            'stream': anthropic_request.get('stream', False),
            'temperature': anthropic_request.get('temperature', 0.7)
        }
    
    def _convert_message(self, msg: dict) -> dict:
        role = msg['role']
        content = msg['content']
        
        if role == 'tool':
            return {
                'role': 'tool',
                'tool_call_id': msg.get('tool_call_id'),
                'content': self._stringify_content(content)
            }
        
        if isinstance(content, list):
            text_parts = []
            tool_calls = []
            
            for part in content:
                if part['type'] == 'text':
                    text_parts.append(part['text'])
                elif part['type'] == 'tool_use':
                    tool_calls.append({
                        'id': part['id'],
                        'type': 'function',
                        'function': {
                            'name': part['name'],
                            'arguments': json.dumps(part['input'])
                        }
                    })
            
            openai_msg = {
                'role': 'assistant' if role == 'assistant' else role,
                'content': '\n'.join(text_parts) if text_parts else None
            }
            
            if tool_calls:
                openai_msg['tool_calls'] = tool_calls
            
            return openai_msg
        
        return {'role': role, 'content': content}
    
    def _convert_tool(self, tool: dict) -> dict:
        return {
            'type': 'function',
            'function': {
                'name': tool['name'],
                'description': tool['description'],
                'parameters': tool['input_schema']
            }
        }
    
    def translate_response(self, openai_response: dict) -> dict:
        choice = openai_response['choices'][0]
        message = choice['message']
        
        content = []
        
        if message.get('content'):
            content.append({
                'type': 'text',
                'text': message['content']
            })
        
        if message.get('tool_calls'):
            for tc in message['tool_calls']:
                content.append({
                    'type': 'tool_use',
                    'id': tc['id'],
                    'name': tc['function']['name'],
                    'input': json.loads(tc['function']['arguments'])
                })
        
        return {
            'id': f"msg_{openai_response['id']}",
            'type': 'message',
            'role': 'assistant',
            'content': content,
            'stop_reason': 'tool_use' if message.get('tool_calls') else 'end_turn',
            'usage': {
                'input_tokens': openai_response['usage']['prompt_tokens'],
```

```python
class ToolRegistry:
    def __init__(self):
        self.tools: dict[str, Tool] = {}
        self.policies: dict[str, SafetyPolicy] = {}
    
    def register(self, tool: Tool, policy: SafetyPolicy = None):
        self.tools[tool.name] = tool
        self.policies[tool.name] = policy or SafetyPolicy.default()
    
    async def execute(self, tool_call: ToolCall, context: ExecutionContext) -> ToolResult:
        tool = self.tools.get(tool_call.name)
        if not tool:
            return ToolResult(error=f"Unknown tool: {tool_call.name}")
        
        policy = self.policies[tool_call.name]
        risk_score = policy.assess_risk(tool_call, context)
        
        if risk_score > policy.auto_approve_threshold:
            approved = await self.request_approval(tool_call, risk_score)
            if not approved:
                return ToolResult(error="User rejected tool execution")
        
        try:
            result = await asyncio.wait_for(
                tool.execute(tool_call.input, context),
                timeout=policy.timeout_seconds
            )
            return ToolResult(output=result)
        except asyncio.TimeoutError:
            return ToolResult(error=f"Tool execution timed out")
        except Exception as e:
            return ToolResult(error=str(e))

class ReadFileTool(Tool):
    name = "read_file"
    description = "Read the contents of a file"
    
    async def execute(self, input: dict, context: ExecutionContext) -> str:
        path = Path(input['path']).resolve()
        
        if not self._is_within_workspace(path, context.workspace_root):
            raise PermissionError("Path outside workspace")
        
        start_line = input.get('start_line', 1)
        end_line = input.get('end_line')
        
        with open(path, 'r') as f:
            lines = f.readlines()
        
        if end_line:
            lines = lines[start_line-1:end_line]
        else:
            lines = lines[start_line-1:]
        
        return ''.join(lines)

class EditFileTool(Tool):
    name = "edit_file"
    description = "Apply a diff to a file"
    
    async def execute(self, input: dict, context: ExecutionContext) -> str:
        path = Path(input['path']).resolve()
        old_string = input['old_string']
        new_string = input['new_string']
        
        with open(path, 'r') as f:
            content = f.read()
        
        if old_string not in content:
            raise ValueError("old_string not found in file")
        
        new_content = content.replace(old_string, new_string, 1)
        
        backup_path = path.with_suffix(path.suffix + '.bak')
        shutil.copy2(path, backup_path)
        
        with open(path, 'w') as f:
            f.write(new_content)
        
        diff = self._generate_diff(content, new_content, str(path))
        return f"File edited successfully.\nDiff:\n{diff}"

class ExecuteCommandTool(Tool):
    name = "execute_command"
    description = "Execute a shell command"
    
    BLOCKED_PATTERNS = [
        r'rm\s+-rf\s+/',
        r'>\s*/dev/sda',
        r'curl\s+.*\|\s*sh',
        r'wget\s+.*\|\s*sh',
        r':\(\)\{ :\|:\& \};:',
    ]
    
    async def execute(self, input: dict, context: ExecutionContext) -> str:
        command = input['command']
        
        for pattern in self.BLOCKED_PATTERNS:
            if re.search(pattern, command):
                raise SecurityError(f"Command matches blocked pattern: {pattern}")
        
        proc = await asyncio.create_subprocess_shell(
            command,
            stdout=asyncio.subprocess.PIPE,
            stderr=asyncio.subprocess.PIPE,
            cwd=context.workspace_root,
            env=self._get_sanitized_env()
        )
        
        stdout, stderr = await proc.communicate()
        return f"Exit code: {proc.returncode}\n\nSTDOUT:\n{stdout.decode()}\n\nSTDERR:\n{stderr.decode()}"
```

```python
class MCPServerManager:
    def __init__(self, config_path: Path):
        self.servers: dict[str, MCPServer] = {}

        all_tools = []

            all_tools.extend(tools)
        return all_tools

        server = self.servers.get(server_name)
```

```python
import asyncio
import tempfile
from pathlib import Path
from dataclasses import dataclass

@dataclass
class SandboxConfig:
    workspace_dir: Path
    allowed_commands: list[str]
    timeout_seconds: float = 30.0
    max_output_bytes: int = 1024 * 1024

class SecureExecutor:
    async def execute(self, command: str, config: SandboxConfig) -> ExecutionResult:
        with tempfile.TemporaryDirectory() as tmpdir:
            # Use bubblewrap/firejail on Linux, seatbelt on macOS
            sandbox_cmd = self._build_sandbox_command(command, config, Path(tmpdir))
            
            proc = await asyncio.create_subprocess_shell(
                sandbox_cmd,
                stdout=asyncio.subprocess.PIPE,
                stderr=asyncio.subprocess.PIPE,
                cwd=config.workspace_dir,
                limit=config.max_output_bytes
            )
            
            try:
                stdout, stderr = await asyncio.wait_for(
                    proc.communicate(),
                    timeout=config.timeout_seconds
                )
                return ExecutionResult(
                    returncode=proc.returncode,
                    stdout=stdout.decode('utf-8', errors='replace'),
                    stderr=stderr.decode('utf-8', errors='replace')
                )
            except asyncio.TimeoutError:
                proc.kill()
                return ExecutionResult(
                    returncode=-1,
                    error=f"Command timed out after {config.timeout_seconds}s"
```

```python
        self.collection = self.client.get_or_create_collection(
            name="code_embeddings",

        for i, chunk in enumerate(chunks):
            self.collection.add(
                documents=[chunk],
                metadatas=[{
                    'file_path': str(file_path),
                    'chunk_index': i,
                    'language': file_path.suffix
                }],
                ids=[f"{file_path}:{i}"]
            )
    
    async def search(self, query: str, top_k: int = 10) -> list[SearchResult]:

            SearchResult(
                file_path=meta['file_path'],
                chunk=doc,
                distance=dist
            )
            for doc, meta, dist in zip(
                results['documents'][0],
                results['metadatas'][0],
                results['distances'][0]
```
