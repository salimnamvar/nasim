# Resolve lib dir relative to this script (works when sourced or exec'd)
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
NASIM_LIB_DIR="$SCRIPT_DIR/../lib/nasim"

# --- Minimal shared utilities (available to all modules) ---

# Args: A message string to output
# Returns: Nothing
# Side effects: Prints the message to stderr and exits with code 1
die() { echo "nasim: $*" >&2; exit 1; }

# Args: A message string to log
# Returns: Nothing
# Side effects: Prints the message to stderr
log() { echo "nasim: $*" >&2; }

# Args: Command name as a string
# Returns: True (0) if command exists, false (1) otherwise
# Side effects: Checks for the presence of the command in the system
have() { command -v "$1" >/dev/null 2>&1; }

# Args: An optional string parameter to check against *dry*
# Returns: True (0) if dry-run mode is active, false (1) otherwise
# Side effects: Evaluates environment variables and input for dry-run conditions
is_dry() { [[ "${NASIM_TEST_MODE:-}${1:-}" == *dry* || "${NASIM_DRY_RUN:-}" == "1" ]]; }

# Hand off to the real command handler
nasim_main "$@"
